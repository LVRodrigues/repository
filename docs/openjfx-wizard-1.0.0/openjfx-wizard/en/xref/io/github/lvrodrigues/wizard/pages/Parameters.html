1   package io.github.lvrodrigues.wizard.pages;
2   
3   import io.github.lvrodrigues.wizard.Constants;
4   import javafx.fxml.FXML;
5   import javafx.scene.control.Spinner;
6   import javafx.scene.control.SpinnerValueFactory;
7   import javafx.scene.control.TextField;
8   
9   /**
10  * Coleta de parâmetros.
11  *
12  * @author $Criado por: $
13  * @author $Alterado por: $
14  * @branch $Ramo: main$
15  */
16  public class Parameters extends AbstractPage {
17  
18      /**
19       * Valor padrão para o parâmetro <b>name</b> quando este ainda não
20       * estiver atribuído.
21       */
22      private static final String PARAM_NAME_DEFAULT = "";
23  
24      /**
25       * Valor padrão para o parâmetro <b>counter</b> quando este ainda
26       * não estiver atribuído.
27       *
28       * <p>Será também o valor mínimo para o componente.
29       */
30      private static final Integer PARAM_COUNTER_DEFAULT = Integer.valueOf("1000");
31  
32      /**
33       * Parâmetro nome para ser utilizado pelas próximas páginas.
34       */
35      @FXML
36      private TextField fieldName;
37  
38      /**
39       * Parêmtro de contagem de execuções.
40       */
41      @FXML
42      private Spinner<Integer> fieldCounter;
43  
44      @Override
45      public String getTitle() {
46          return "Parâmetros";
47      }
48  
49      @Override
50      public void onHide() {
51          datas().put(Constants.PARAM_NAME, fieldName.getText());
52          datas().put(Constants.PARAM_COUNTER, fieldCounter.getValue().toString());
53          super.onHide();
54      }
55  
56      @Override
57      public void onShow() {
58          super.onShow();
59          fieldName.setText(datas().getOrDefault(Constants.PARAM_NAME, PARAM_NAME_DEFAULT));
60          fieldCounter.getValueFactory().setValue(
61              Integer.valueOf(datas().getOrDefault(
62                  Constants.PARAM_COUNTER, PARAM_COUNTER_DEFAULT.toString())));
63      }
64  
65      @Override
66      public boolean canDoNext() {
67          return !fieldName.getText().isBlank();
68      }
69  
70      /**
71       * Execução automática na inicialização do componente.
72       */
73      @FXML
74      public void initialize() {
75          fieldCounter.setValueFactory(
76              new SpinnerValueFactory.IntegerSpinnerValueFactory(
77                  PARAM_COUNTER_DEFAULT, Integer.MAX_VALUE));
78      }
79  }
